    1Simple Usage

    Given you have an Entity named source/entity/Foo.js already in place, rename the Font to source/entity/Foo.fnt.

    After that you can access the Font in the code: 

    lychee.define('game.entity.Foo').exports(function(lychee, game, global, attachments) {

  var font = attachments['fnt'];
  if (font instanceof Font) {
    console.log('Yay, font available!', font);
  }

});

    2Advanced Usage

    Given you have an Entity named source/entity/Foo.js already in place, you can rename the Fonts to source/entity/Foo.label.fnt and source/entity/Foo.small.fnt.

    After that you can access the Fonts in the code: 

    lychee.define('game.entity.Foo').exports(function(lychee, game, global, attachments) {

  var fonts = {
    label: attachments['label.fnt'],
    small: attachments['small.fnt']
  };

  if (fonts.label instanceof Font) {
    console.log('Yay, label font available!', fonts.label);
  }

  if (fonts.small instanceof Font) {
    console.log('Yay, small font available!', fonts.small);
  }

});
